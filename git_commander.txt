[commit]
git add <file1> <file2>
git commit -m<comment>
git commit -am<comment> // 修改或者删除的文件，可以一起操作 add + commit

[push]
git push origin - <branch>  // 首次push一个分支，设置关联
git push 

[pull]
git pull // 更新

[branch]
git branch -vv // 查看本地分支，同时显示关联关系
git branch -a  // 查看所有分支，本地 + 远程
git checkout -b <branch_name>  // 新建一个分支，并切换过去
git branch -d <branch_name>    // 删除本地分支
git push origin :<branch_name> // 删除远程分支
